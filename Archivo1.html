Tu Peso en otro Planeta

En La Tierra pesas distinto que en Marte o Júpiter

var usuario = prompt("Cual Es Tu Peso?"); var planeta = parseInt(prompt("Elije tu planeta. 1 es Marte, 2 es Jupiter")); var peso = parseInt(usuario); var g_tierra = 9.8; var g_marte = 3.7; var g_jupiter = 24.8; var peso_final; if (planeta == 1) { peso_final = peso * g_marte / g_tierra; } else if(planeta == 2) { peso_final = peso * g_jupiter / g_tierra; } else { peso_final = "1000000"; } peso_final = parseInt(peso_final); if (planeta == 1) { document.write("Tu Peso en Marte es " + peso_final + " Kilos"); } else if(planeta == 2) { document.write("Tu Peso en Júpiter es " + peso_final + " Kilos"); } else { document.write("Tu nivel de Taradez es de: " + peso_final + " Taradillos"); }
